function showDir($dirs=__DIR__)
{
	$dir = opendir($dirs);
	while ($f = readdir($dir)) {
		if (! in_array($f,['.','..'])) {
			$file = $dirs.'/'.$f;
			if (is_dir($file)) {
				showDir($file);
			}else{
				echo '文件名：'.$file.'<br />';
			}
		}
	}
	closedir($dir);
}

function getfiles($path){ 
	foreach(scandir($path) as $afile)
	{
		if($afile !='.' && $afile!='..'){
			$dir = $path.'/'.$afile;
			if(is_dir($path.'/'.$afile)) 
			{ 
				getfiles($path.'/'.$afile.'/'); 
			} else { 
				echo $dir.'<br>'; 
			} 
		}
	} 
}


function getfiles2($path=__DIR__){ 
	foreach(glob($path . '/*') as $afile){ 
		if(is_dir($afile)) { 
			getfiles($afile); 
		} else {
			echo $afile.'<br />';
		} 
	} 
}
