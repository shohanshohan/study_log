
for($i=1; $i<9; $i++){
  for($j=1; $j<$i; $j++){
    $num = $j.'x'.$i.'='.$j*$i;
    echo $num.'&nbsp;&nbsp;&nbsp;';
  }
  echo '<br />';
}
